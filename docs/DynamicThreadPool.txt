package config;

import java.util.concurrent.*;

public class DynamicThreadPool {

    private DynamicThreadPool() {
        throw new UnsupportedOperationException("DynamicThreadPool cannot be instantiated");
    }

    public static ExecutorService create() {
        int corePoolSize = 10;
        int maxPoolSize = 100;
        long keepAliveTime = 60L; // seconds
        BlockingQueue<Runnable> queue = new LinkedBlockingQueue<>(1000);


        return new ThreadPoolExecutor(
                corePoolSize,
                maxPoolSize,
                keepAliveTime,
                TimeUnit.SECONDS,
                queue
        );
    }
}
